546 /** \brief wstawianie kuli cząstek na życzenie
547  * \return void
548  */
549 void addSphere()
550 {
551  // inject a sphere of particles
552  float pr = psystem->getParticleRadius();
553  float tr = pr+(pr*2.0f)*ballr;
554  float pos[4], vel[4];
555  pos[0] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
556  pos[1] = 1.0f - tr;
557  pos[2] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
558  pos[3] = 0.0f;
559  //vel[0] = vel[1] = vel[2] = vel[3] = 0.0f;
560  vel[0]=frand()/100.0f;
561  vel[1]=frand()/100.0f;
562  vel[2]=frand()/100.0f;
563  vel[3]=frand()/100.0f;
564  psystem->addSphere(0, pos, vel, ballr, pr*2.0f);
565 }

645 // transfrom vector by matrix
646 /** \brief transfrom vector by matrix
647  * \param v float*
648  * \param r float*
649  * \param m GLfloat*
650  * \return void
651  */
652 void xform(float *v, float *r, GLfloat *m)
653 {
654  r[0] = v[0]*m[0] + v[1]*m[4] + v[2]*m[8] + m[12];
655  r[1] = v[0]*m[1] + v[1]*m[5] + v[2]*m[9] + m[13];
656  r[2] = v[0]*m[2] + v[1]*m[6] + v[2]*m[10] + m[14];
657 }
