183 void runBenchmark(int iterations, char *exec_path)
184 {
185  printf("Run %u particles simulation for %d iterations...\n\n", numParticles, iterations);
186  cudaDeviceSynchronize();
187  sdkStartTimer(&timer);
188 
189  for (int i = 0; i < iterations; ++i)
190  {
191  psystem->update(timestep);
192  }
193 
194  cudaDeviceSynchronize();
195  sdkStopTimer(&timer);
196  float fAvgSeconds = ((float)1.0e-3 * (float)sdkGetTimerValue(&timer)/(float)iterations);
197 
198  printf("particles, Throughput = %.4f KParticles/s, Time = %.5f s, Size = %u particles, NumDevsUsed = %u, Workgroup = %u\n",
199  (1.0e-3 * numParticles)/fAvgSeconds, fAvgSeconds, numParticles, 1, 0);
200 
201  if (g_refFile)
202  {
203  printf("\nChecking result...\n\n");
204  float *hPos = (float *)malloc(sizeof(float)*4*psystem->getNumParticles());
205  copyArrayFromDevice(hPos, psystem->getCudaPosVBO(),
206  0, sizeof(float)*4*psystem->getNumParticles());
207 
208  sdkDumpBin((void *)hPos, sizeof(float)*4*psystem->getNumParticles(), "particles.bin");
209 
210  if (!sdkCompareBin2BinFloat("particles.bin", g_refFile, sizeof(float)*4*psystem->getNumParticles(),
211  MAX_EPSILON_ERROR, THRESHOLD, exec_path))
212  {
213  g_TotalErrors++;

237 void display()
238 {
239  sdkStartTimer(&timer);
240 
241  // update the simulation
242  if (!bPause)
243  {
244  psystem->setIterations(iterations);
245  psystem->setDamping(damping);
246  psystem->setGravity(-gravity);
247  psystem->setCollideSpring(collideSpring);
248  psystem->setCollideDamping(collideDamping);
249  psystem->setCollideShear(collideShear);
250  psystem->setCollideAttraction(collideAttraction);
251 
252  licznik++;
253  bigRadius=bigRadius0-A*sqrt(licznik*timestep);//r=r0-A*sqrt(t)
254  psystem->setBigRadius(bigRadius);
255 
256  psystem->update(timestep);
257 
258  if (renderer)
259  {
260  renderer->setVertexBuffer(psystem->getCurrentReadBuffer(), psystem->getNumParticles());
261  }
262  }
263 
264  // render
265  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
266 
267  // view transform
268  glMatrixMode(GL_MODELVIEW);
269  glLoadIdentity();
270 
271  for (int c = 0; c < 3; ++c)
272  {
273  camera_trans_lag[c] += (camera_trans[c] - camera_trans_lag[c]) * inertia;
274  camera_rot_lag[c] += (camera_rot[c] - camera_rot_lag[c]) * inertia;
275  }
276 
277  glTranslatef(camera_trans_lag[0], camera_trans_lag[1], camera_trans_lag[2]);
278  glRotatef(camera_rot_lag[0], 1.0, 0.0, 0.0);
279  glRotatef(camera_rot_lag[1], 0.0, 1.0, 0.0);
280 
281  glGetFloatv(GL_MODELVIEW_MATRIX, modelView);
282 
283  // cube -> sphere
284  glColor3f(1.0, 1.0, 1.0);
285  //glutWireCube(2.0);
286  glutWireSphere(bigRadius, 20, 10);
287 
288  // collider
289  glPushMatrix();
290  float3 p = psystem->getColliderPos();
291  glTranslatef(p.x, p.y, p.z);
292  glColor3f(1.0, 0.0, 0.0);
293  glutSolidSphere(psystem->getColliderRadius(), 20, 10);
294  glPopMatrix();
295 
296  if (renderer && displayEnabled)
297  {
298  renderer->display(displayMode);
299  }
300 
301  if (displaySliders)
302  {
303  glDisable(GL_DEPTH_TEST);
304  glBlendFunc(GL_ONE_MINUS_DST_COLOR, GL_ZERO); // invert color
305  glEnable(GL_BLEND);
306  params->Render(0, 0);
307  glDisable(GL_BLEND);
308  glEnable(GL_DEPTH_TEST);
309  }
310 
311  sdkStopTimer(&timer);
312 
313  glutSwapBuffers();
314  glutReportErrors();
315 
316  computeFPS();
317 }

324 void addSphere()
325 {
326  // inject a sphere of particles
327  float pr = psystem->getParticleRadius();
328  float tr = pr+(pr*2.0f)*ballr;
329  float pos[4], vel[4];
330  pos[0] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
331  pos[1] = 1.0f - tr;
332  pos[2] = -1.0f + tr + frand()*(2.0f - tr*2.0f);
333  pos[3] = 0.0f;
334  //vel[0] = vel[1] = vel[2] = vel[3] = 0.0f;
335  vel[0]=frand()/100.0f;
336  vel[1]=frand()/100.0f;
337  vel[2]=frand()/100.0f;
338  vel[3]=frand()/100.0f;
339  psystem->addSphere(0, pos, vel, ballr, pr*2.0f);
340 }

400 // transfrom vector by matrix
401 void xform(float *v, float *r, GLfloat *m)
402 {
403  r[0] = v[0]*m[0] + v[1]*m[4] + v[2]*m[8] + m[12];
404  r[1] = v[0]*m[1] + v[1]*m[5] + v[2]*m[9] + m[13];
405  r[2] = v[0]*m[2] + v[1]*m[6] + v[2]*m[10] + m[14];
406 }
407 
408 // transform vector by transpose of matrix
409 void ixform(float *v, float *r, GLfloat *m)
410 {
411  r[0] = v[0]*m[0] + v[1]*m[1] + v[2]*m[2];
412  r[1] = v[0]*m[4] + v[1]*m[5] + v[2]*m[6];
413  r[2] = v[0]*m[8] + v[1]*m[9] + v[2]*m[10];
414 }

426 void motion(int x, int y)
427 {
428  float dx, dy;
429  dx = (float)(x - ox);
430  dy = (float)(y - oy);
431 
432  if (displaySliders)
433  {
434  if (params->Motion(x, y))
435  {
436  ox = x;
437  oy = y;
438  glutPostRedisplay();
439  return;
440  }
441  }
442 
443  switch (mode)
444  {
445  case M_VIEW:
446  if (buttonState == 3)
447  {
448  // left+middle = zoom
449  camera_trans[2] += (dy / 100.0f) * 0.5f * fabs(camera_trans[2]);
450  }
451  else if (buttonState & 2)
452  {
453  // middle = translate
454  camera_trans[0] += dx / 100.0f;
455  camera_trans[1] -= dy / 100.0f;
456  }
457  else if (buttonState & 1)
458  {
459  // left = rotate
460  camera_rot[0] += dy / 5.0f;
461  camera_rot[1] += dx / 5.0f;
462  }
463 
464  break;
465 
466  case M_MOVE:
467  {
468  float translateSpeed = 0.003f;
469  float3 p = psystem->getColliderPos();
470 
471  if (buttonState==1)
472  {
473  float v[3], r[3];
474  v[0] = dx*translateSpeed;
475  v[1] = -dy*translateSpeed;
476  v[2] = 0.0f;
477  ixform(v, r, modelView);
478  p.x += r[0];
479  p.y += r[1];
480  p.z += r[2];
481  }
482  else if (buttonState==2)
483  {
484  float v[3], r[3];
485  v[0] = 0.0f;
486  v[1] = 0.0f;
487  v[2] = dy*translateSpeed;
488  ixform(v, r, modelView);
489  p.x += r[0];
490  p.y += r[1];
491  p.z += r[2];
492  }
493 
494  psystem->setColliderPos(p);
495  }
496  break;
497  }
498 
499  ox = x;
500  oy = y;
501 
502  demoMode = false;
503  idleCounter = 0;
504 
505  glutPostRedisplay();
506 }
