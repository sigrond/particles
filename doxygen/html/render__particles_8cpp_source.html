55 void ParticleRenderer::_drawPoints()
56 {
57  if (!m_vbo)
58  {
59  glBegin(GL_POINTS);
60  {
61  int k = 0;
62 
63  for (int i = 0; i < m_numParticles; ++i)
64  {
65  glVertex3fv(&m_pos[k]);
66  k += 4;
67  }
68  }
69  glEnd();
70  }
71  else
72  {
73  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_vbo);
74  glVertexPointer(4, GL_FLOAT, 0, 0);
75  glEnableClientState(GL_VERTEX_ARRAY);
76 
77  if (m_colorVBO)
78  {
79  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_colorVBO);
80  glColorPointer(4, GL_FLOAT, 0, 0);
81  glEnableClientState(GL_COLOR_ARRAY);
82  }
83 
84  glDrawArrays(GL_POINTS, 0, m_numParticles);
85 
86  glBindBufferARB(GL_ARRAY_BUFFER_ARB, 0);
87  glDisableClientState(GL_VERTEX_ARRAY);
88  glDisableClientState(GL_COLOR_ARRAY);
89  }
90 }

92 void ParticleRenderer::display(DisplayMode mode /* = PARTICLE_POINTS */)
93 {
94  switch (mode)
95  {
96  case PARTICLE_POINTS:
97  glColor3f(1, 1, 1);
98  glPointSize(m_pointSize);
99  _drawPoints();
100  break;
101 
102  default:
103  case PARTICLE_SPHERES:
104  glEnable(GL_POINT_SPRITE_ARB);
105  glTexEnvi(GL_POINT_SPRITE_ARB, GL_COORD_REPLACE_ARB, GL_TRUE);
106  glEnable(GL_VERTEX_PROGRAM_POINT_SIZE_NV);
107  glDepthMask(GL_TRUE);
108  glEnable(GL_DEPTH_TEST);
109 
110  glUseProgram(m_program);
111  glUniform1f(glGetUniformLocation(m_program, "pointScale"), zoom*m_window_h / tanf(m_fov*0.5f*(float)M_PI/180.0f));
112  glUniform1f(glGetUniformLocation(m_program, "pointRadius"), m_particleRadius);
113 
114  glColor3f(1, 1, 1);
115  _drawPoints();
116 
117  glUseProgram(0);
118  glDisable(GL_POINT_SPRITE_ARB);
119  break;
120  }
121 }

124 ParticleRenderer::_compileProgram(const char *vsource, const char *fsource)
125 {
126  GLuint vertexShader = glCreateShader(GL_VERTEX_SHADER);
127  GLuint fragmentShader = glCreateShader(GL_FRAGMENT_SHADER);
128 
129  glShaderSource(vertexShader, 1, &vsource, 0);
130  glShaderSource(fragmentShader, 1, &fsource, 0);
131 
132  glCompileShader(vertexShader);
133  glCompileShader(fragmentShader);
134 
135  GLuint program = glCreateProgram();
136 
137  glAttachShader(program, vertexShader);
138  glAttachShader(program, fragmentShader);
139 
140  glLinkProgram(program);
141 
142  // check if program linked
143  GLint success = 0;
144  glGetProgramiv(program, GL_LINK_STATUS, &success);
145 
146  if (!success)
147  {
148  char temp[256];
149  glGetProgramInfoLog(program, 256, 0, temp);
150  printf("Failed to link program:\n%s\n", temp);
151  glDeleteProgram(program);
152  program = 0;
153  }
154 
155  return program;
156 }
