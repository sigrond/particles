157 // calculate position in uniform grid
158 __device__ int3 calcGridPos(float3 p)
159 {
160  int3 gridPos;
161  gridPos.x = floor((p.x - params.worldOrigin.x) / params.cellSize.x);
162  gridPos.y = floor((p.y - params.worldOrigin.y) / params.cellSize.y);
163  gridPos.z = floor((p.z - params.worldOrigin.z) / params.cellSize.z);
164  return gridPos;
165 }

167 // calculate address in grid from position (clamping to edges)
168 __device__ uint calcGridHash(int3 gridPos)
169 {
170  gridPos.x = gridPos.x & (params.gridSize.x-1); // wrap grid, assumes size is power of 2
171  gridPos.y = gridPos.y & (params.gridSize.y-1);
172  gridPos.z = gridPos.z & (params.gridSize.z-1);
173  return __umul24(__umul24(gridPos.z, params.gridSize.y), params.gridSize.x) + __umul24(gridPos.y, params.gridSize.x) + gridPos.x;
174 }

178 void calcHashD(uint *gridParticleHash, // output
179  uint *gridParticleIndex, // output
180  float4 *pos, // input: positions
181  uint numParticles)
182 {

184 
185  if (index >= numParticles) return;

187  volatile float4 p = pos[index];

198 // rearrange particle data into sorted order, and find the start of each cell
199 // in the sorted hash array
200 __global__
201 void reorderDataAndFindCellStartD(uint *cellStart, // output: cell start index
202  uint *cellEnd, // output: cell end index
203  float4 *sortedPos, // output: sorted positions
204  float4 *sortedVel, // output: sorted velocities
205  uint *gridParticleHash, // input: sorted grid hashes
206  uint *gridParticleIndex,// input: sorted particle indices
207  float4 *oldPos, // input: sorted position array
208  float4 *oldVel, // input: sorted velocity array
209  uint numParticles)
210 {
211  extern __shared__ uint sharedHash[]; // blockSize + 1 elements
212  uint index = __umul24(blockIdx.x,blockDim.x) + threadIdx.x;
213 
214  uint hash;
215 
216  // handle case when no. of particles not multiple of block size
217  if (index < numParticles)
218  {
219  hash = gridParticleHash[index];
220 
221  // Load hash data into shared memory so that we can look
222  // at neighboring particle's hash value without loading
223  // two hash values per thread
224  sharedHash[threadIdx.x+1] = hash;
225 
226  if (index > 0 && threadIdx.x == 0)
227  {
228  // first thread in block must load neighbor particle hash
229  sharedHash[0] = gridParticleHash[index-1];
230  }
231  }
232 
233  __syncthreads();
234 
235  if (index < numParticles)
236  {
237  // If this particle has a different cell index to the previous
238  // particle then it must be the first particle in the cell,
239  // so store the index of this particle in the cell.
240  // As it isn't the first particle, it must also be the cell end of
241  // the previous particle's cell
242 
243  if (index == 0 || hash != sharedHash[threadIdx.x])
244  {
245  cellStart[hash] = index;
246 
247  if (index > 0)
248  cellEnd[sharedHash[threadIdx.x]] = index;
249  }
250 
251  if (index == numParticles - 1)
252  {
253  cellEnd[hash] = index + 1;
254  }
255 
256  // Now use the sorted index to reorder the pos and vel data
257  uint sortedIndex = gridParticleIndex[index];
258  float4 pos = FETCH(oldPos, sortedIndex); // macro does either global read or texture fetch
259  float4 vel = FETCH(oldVel, sortedIndex); // see particles_kernel.cuh
260 
261  sortedPos[index] = pos;
262  sortedVel[index] = vel;
263  }

323 // collide a particle against all other particles in a given cell
324 __device__
325 float3 collideCell(int3 gridPos,
326  uint index,
327  float3 pos,
328  float3 vel,
329  float4 *oldPos,
330  float4 *oldVel,
331  uint *cellStart,
332  uint *cellEnd)
333 {
334  uint gridHash = calcGridHash(gridPos);
335 
336  // get start of bucket for this cell
337  uint startIndex = FETCH(cellStart, gridHash);
338 
339  float3 force = make_float3(0.0f);
340 
341  if (startIndex != 0xffffffff) // cell is not empty
342  {
343  // iterate over particles in this cell
344  uint endIndex = FETCH(cellEnd, gridHash);
345 
346  for (uint j=startIndex; j<endIndex; j++)
347  {
348  if (j != index) // check not colliding with self
349  {
350  float3 pos2 = make_float3(FETCH(oldPos, j));
351  float3 vel2 = make_float3(FETCH(oldVel, j));
352 
353  // collide two spheres
354  force += collideSpheres(pos, pos2, vel, vel2, params.particleRadius, params.particleRadius, params.attraction);
355  }
356  }
357  }
358 
359  return force;
