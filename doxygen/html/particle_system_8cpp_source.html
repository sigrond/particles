113 void colorRamp(float t, float *r)
114 {
115  const int ncolors = 7;
116  float c[ncolors][3] =
117  {
118  { 1.0, 0.0, 0.0, },
119  { 1.0, 0.5, 0.0, },
120  { 1.0, 1.0, 0.0, },
121  { 0.0, 1.0, 0.0, },
122  { 0.0, 1.0, 1.0, },
123  { 0.0, 0.0, 1.0, },
124  { 1.0, 0.0, 1.0, },
125  };
126  t = t * (ncolors-1);
127  int i = (int) t;
128  float u = t - floor(t);
129  r[0] = lerp(c[i][0], c[i+1][0], u);
130  r[1] = lerp(c[i][1], c[i+1][1], u);
131  r[2] = lerp(c[i][2], c[i+1][2], u);

511 ParticleSystem::addSphere(int start, float *pos, float *vel, int r, float spacing)
512 {
513  uint index = start;
514 
515  for (int z=-r; z<=r; z++)
516  {
517  for (int y=-r; y<=r; y++)
518  {
519  for (int x=-r; x<=r; x++)
520  {
521  float dx = x*spacing;
522  float dy = y*spacing;
523  float dz = z*spacing;
524  float l = sqrtf(dx*dx + dy*dy + dz*dz);
525  float jitter = m_params.particleRadius*0.01f;
526 
527  if ((l <= m_params.particleRadius*2.0f*r) && (index < m_numParticles))
528  {
529  m_hPos[index*4] = pos[0] + dx + (frand()*2.0f-1.0f)*jitter;
530  m_hPos[index*4+1] = pos[1] + dy + (frand()*2.0f-1.0f)*jitter;
531  m_hPos[index*4+2] = pos[2] + dz + (frand()*2.0f-1.0f)*jitter;
532  m_hPos[index*4+3] = pos[3];
533 
534  m_hVel[index*4] = vel[0];
535  m_hVel[index*4+1] = vel[1];
536  m_hVel[index*4+2] = vel[2];

538  index++;
539  }
